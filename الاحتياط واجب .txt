                    dispose();

        if(AllUsers.getSelectedItem()=="Admin"){

        this.Amount = Amount;
        this.Region = Region;
        this.paid=paid;
    }

    public void setPaid(boolean paid) {
        this.paid = paid;
    }

        return this.paid;
    }

    public void setCo_ssn(String Customer_ssn) {
        this.Customer_ssn = Customer_ssn;

    }

    public void setAmount(Double Amount) {
        this.Amount = Amount;
    }

    public void setRegion(String Region) {
        this.Region = Region;
    }

    public Double getAmount() {
        return Amount;
    }

    

    public String getCustomer_ssn() {

    }

    public List<Bills> get_all_bills() {
        
        List<Bills> bill = new ArrayList<Bills>();
        try {
            
            File f = new File("E:\\projects java\\FileHandler\\bill.txt");
            Scanner sc = new Scanner(f);
            
            while (sc.hasNextLine()) {
                String line = sc.nextLine();
                
                String[] details = line.split(",");                
                
                
                String Meter_Code = details[0];
                String Cus_Ssn = details[1];
                double Amount = Double.parseDouble(details[2]);
                String region = details[3];
                boolean paid = Boolean.parseBoolean(details[4]);
                Bills p = new Bills();
                
                
                p.setMeter_code(Meter_Code);
                p.setCo_ssn(Cus_Ssn);
                p.setAmount(Amount);
                p.setRegion(region);
                p.setPaid(paid);
                
                bill.add(p);                
            }
            sc.close();
            /*
    for(Bills p: bill){
        System.out.println(p.toString());
    }
             */
            
        } catch (FileNotFoundException e) {            
            e.printStackTrace();
        }
        return bill;
        
    }
    
    public void Update_Bills(List<Bills> bill) {
        
        try {
            
            FileWriter fw = new FileWriter("E:\\projects java\\FileHandler\\bill.txt", false);

public class Old_Customer extends Customer {

    FileHandler f1 = new FileHandler();
    // private ArrayList<Bills> paidBills;
    private Meter houseMeter = new Meter();
    private String Complain;

    // Constructors
    public Old_Customer() {

    }

    public Old_Customer(String ssn, String name, String region, String address, String email, String meterCode, String UserName, String Password) {
        super(ssn, name, region, address, email);
        this.houseMeter.meterCode = meterCode;
        this.setPassword(Password);
        this.setUsername(UserName);

    }

    // Mutators
    public void setComplain(String complain) {
        this.Complain = complain;
    }

    public void setMeterCode(String meterCode) {
        this.houseMeter.meterCode = meterCode;
    }

    // Accessors
    public String getComplain() {
        return Complain;
    }

    public String getMeterCode() {
        return houseMeter.meterCode;
    }

    public String toString() {
        return getName() + "," + this.getSsn() + "," + this.getRegion() + "," + this.getAddress()
                + "," + this.getEmail() + "," + this.houseMeter.meterCode + "," + this.getUsername() + "," + this.getPassword() + "\n";
    }

    public boolean add_customer(Old_Customer Oc) {

        ArrayList<Old_Customer> customer_list = new ArrayList<Old_Customer>();

        customer_list = (ArrayList<Old_Customer>) f1.get_all_Customers();
        for (Old_Customer c : customer_list) {
            if (Oc.getSsn().equals(c.getSsn()) || Oc.getEmail().equals(c.getEmail()) || Oc.getMeterCode().equals(c.getMeterCode()) || Oc.getUsername().equals(c.getUsername()) || Oc.getPassword().equals(c.getPassword())) {
                return false;
            }
        }
        customer_list.add(Oc);
        f1.Update_Customers(customer_list);
        return true;
    }

    public Double BillWithMc(String meter_code) {
        Double sum = 0.0;
        ArrayList<Bills> bill_List = new ArrayList<Bills>();
        bill_List = (ArrayList<Bills>) f1.get_all_bills();
        for (Bills c : bill_List) {
            if (c.getMeter_code().equals(meter_code) && !c.getPaid()) {
                sum += c.getAmount();
            }
        }
        return sum;
    }

    public void pay(String Meter_code) {
        ArrayList<Bills> bill_List = new ArrayList<Bills>();
        bill_List = (ArrayList<Bills>) f1.get_all_bills();
        for (Bills c : bill_List) {
            if (c.getMeter_code().equals(Meter_code) && !c.getPaid()) {
                c.setPaid(true);
            }
        }
        f1.Update_Bills(bill_List);
    }

    public void MonthlyReading(double meter, String mc) {

        ArrayList<Meter> Meter_List = new ArrayList<>();
        FileHandler f1 = new FileHandler();
        Meter_List = (ArrayList<Meter>) f1.get_all_meters();
        for (Meter m : Meter_List) {
            if (m.getMeterCode() == mc) {
                double[] x = m.getMonthlyReading();
                x[0] = x[1];
                x[1] = meter;
            }
        }
    }

}

    private void Customer_RegestrationActionPerformed(java.awt.event.ActionEvent evt) {                                                      
        // TODO add your handling code here:
        String name = Name.getText();
        String ssn = National_ID.getText();
        String email = EMail.getText();
        String region = Region.getText();
        String meter_code = Meter_Code.getText();
        String address = Address.getText();
        String Pass = Password.getText();
        String user = UserName.getText();
        String passReset=PasswordReset.getText();
        if(!Pass.equals(passReset)){
            JOptionPane.showMessageDialog(null, "password not matched with password reset");  
        }
        else{
        Old_Customer x = new Old_Customer(ssn,name,region, address, email, meter_code, user, Pass);
        boolean r= x.add_customer(x);

        if (r) {
            JOptionPane.showMessageDialog(null, "Account is created");
        } else {
            JOptionPane.showMessageDialog(null, "Account is already exist");
        }
        }
        

    }                                                     

    private void UploadActionPerformed(java.awt.event.ActionEvent evt) {                                       
        // TODO add your handling code here:
        JFileChooser chooser = new JFileChooser();
        chooser.showOpenDialog(null);

        File f = chooser.getSelectedFile();
        if (!f.isFile()) {
            JOptionPane.showMessageDialog(null, "it is not an Image");
        }
        String filename = f.getAbsolutePath();
        imageText.setText(filename);

        try {
            ImageIcon icon = new ImageIcon(ImageIO.read(new File(filename)));
            image.setIcon(icon);
        } catch (IOException ex) {
            Logger.getLogger(information.class
                    .getName()).log(Level.SEVERE, null, ex);
        }

    }
